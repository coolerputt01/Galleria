def get(self, request, pk):
    album = get_object_or_404(Album, pk=pk)
    is_liked = album.likes.filter(user=request.user).exists()
    return Response({
        "likes": album.likes.count(),
        "is_liked": is_liked
    }, status=status.HTTP_200_OK)
